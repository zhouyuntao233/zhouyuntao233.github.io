树莓派推流

树莓派推流方式

1.使用自带的raspivid
sudo raspivid -o - -t 0 -n -w 640 -h 360 -t 0 -a 12 -a 1024 -a "PiCam %Y-%m-%d %X" -fps 25|ffmpeg -i - -vcodec copy -an -f flv rtmp://81.70.192.71/dash/video
2.Python 脚本推流
import subprocess as sp
import cv2
rtmpUrl = "rtmp://81.70.192.71:1935/dash/video"

cap = cv2.VideoCapture(0)

# Get video information
fps = int(cap.get(cv2.CAP_PROP_FPS))
width = int(cap.get(cv2.CAP_PROP_FRAME_WIDTH))
height = int(cap.get(cv2.CAP_PROP_FRAME_HEIGHT))

# ffmpeg command
command = ['ffmpeg',
        '-y',
        '-f', 'rawvideo',
        '-vcodec','rawvideo',
        '-pix_fmt', 'bgr24',
        '-s', "{}x{}".format(width, height),
        '-r', str(fps),
        '-i', '-',
        '-c:v', 'libx264',
        '-pix_fmt', 'yuv420p',
        '-preset', 'ultrafast',
        '-f', 'flv', 
        rtmpUrl]

# 管道配置
p = sp.Popen(command, stdin=sp.PIPE)
        
# read webcamera
while(cap.isOpened()):
    ret, frame = cap.read()
    if not ret:
        print("Opening camera is failed")
        break

    p.stdin.write(frame.tobytes())

服务器端配置

服务器端主要配置

编译安装nginx和nginx-rtmp模块

git clone https://github.com/arut/nginx-rtmp-module.gitcd nginx-1.18.0/
cd nginx-1.18.0/
sudo ./configure --with-http_ssl_module --add-module=../nginx-rtmp-module
sudo make 
sudo make install

nginx相关操作

sudo cp objs/nginx /usr/local/nginx/sbin/nginx # 涉及到添加模块如nginx配置ssl证书
sudo /usr/local/nginx/sbin/nginx -c /usr/local/nginx/conf/nginx.conf
sudo cp /usr/local/nginx/sbin/nginx /usr/local/bin/ #把nginx加入到环境变量里
nginx            #第一次启动
nginx -t        #检查配置文件是否正确
nginx -s reload  #平滑重启，修改配置文件后，不断服务重启
nginx -s stop    #停止服务

安卓端播放直播流

安卓端采用Exoplayer播放流媒体

Exoplayer文档

# 引入包
implementation 'com.google.android.exoplayer:exoplayer:2.13.1'

activity_main.xml

<?xml version="1.0" encoding="utf-8"?>
<androidx.constraintlayout.widget.ConstraintLayout xmlns:android="http://schemas.android.com/apk/res/android"
    xmlns:app="http://schemas.android.com/apk/res-auto"
    xmlns:tools="http://schemas.android.com/tools"
    android:layout_width="match_parent"
    android:layout_height="match_parent"
    tools:context=".MainActivity">

    <com.google.android.exoplayer2.ui.PlayerView
        android:id="@+id/video_view"
        android:layout_width="match_parent"
        android:layout_height="match_parent"/>
</androidx.constraintlayout.widget.ConstraintLayout>

MainActivity

public class MainActivity extends AppCompatActivity {

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);
        PlayerView playerView=findViewById(R.id.video_view);
        // Create a data source factory.
        DataSource.Factory dataSourceFactory = new DefaultHttpDataSource.Factory();
        // Create a DASH media source pointing to a DASH manifest uri.
        MediaSource mediaSource =
                new DashMediaSource.Factory(dataSourceFactory)
                        .createMediaSource(MediaItem.fromUri("http://81.70.192.71/dash/video.mpd"));
        // Create a player instance.
        SimpleExoPlayer player = new SimpleExoPlayer.Builder(this).build();
        // Set the media source to be played.
        playerView.setPlayer(player);
        player.setPlayWhenReady(true);
        player.setMediaSource(mediaSource);
        // Prepare the player.
        player.prepare();
    }
}
